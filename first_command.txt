Комманды баша
cd /c/users/user   // сменить рабочую директорию
cd ..    //папка на 1 порядок выше
pwd    // показать рабочую директорию
ls     // содержимое текущей директории
mkdir  // создать новую директорию
hostname   // имя машины
whoami    // ваш логин
ipconfig   // текущие настройки протокола TCP/IP


git clone <адрес>
git status //инфо о текущем состоянии репозитория


git add <имя файла>
git commit -m "месседж"
git push


git branch <название новой ветки>
git checkout <название ветки, на которую нужно переключиться>
git push origin < имя ветки, куда пушим >
